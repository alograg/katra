 41: /**
 42:  * POP Before SMTP Authentication Class
 43:  * Version 5.0.0
 44:  *
 45:  * Author: Richard Davey (rich@corephp.co.uk)
 46:  * Modifications: Andy Prevost
 47:  * License: LGPL, see PHPMailer License
 48:  *
 49:  * Specifically for PHPMailer to allow POP before SMTP authentication.
 50:  * Does not yet work with APOP - if you have an APOP account, contact Richard Davey
 51:  * and we can test changes to this script.
 52:  *
 53:  * This class is based on the structure of the SMTP class originally authored by Chris Ryan
 54:  *
 55:  * This class is rfc 1939 compliant and implements all the commands
 56:  * required for POP3 connection, authentication and disconnection.
 57:  *
 58:  * @package PHPMailer
 59:  * @author Richard Davey
 60:  */
 61: 
 62: class POP3
 63: {
 64:     /**
 65:      * Default POP3 port
 66:      * @var int
 67:      */
 68: 
 69:     public $POP3_PORT = 110;
 70: 
 71:     /**
 72:      * Default Timeout
 73:      * @var int
 74:      */
 75:     public $POP3_TIMEOUT = 30;
 76: 
 77:     /**
 78:      * POP3 Carriage Return + Line Feed
 79:      * @var string
 80:      */
 81:     public $CRLF = "\r\n";
 82: 
 83:     /**
 84:      * Displaying Debug warnings? (0 = now, 1+ = yes)
 85:      * @var int
 86:      */
 87:     public $do_debug = 2;
 88: 
 89:     /**
 90:      * POP3 Mail Server
 91:      * @var string
 92:      */
 93:     public $host;
 94: 
 95:     /**
 96:      * POP3 Port
 97:      * @var int
 98:      */
 99:     public $port;
100: 
101:     /**
102:      * POP3 Timeout Value
103:      * @var int
104:      */
105:     public $tval;
106: 
107:     /**
108:      * POP3 Username
109:      * @var string
110:      */
111:     public $username;
112: 
113:     /**
114:      * POP3 Password
115:      * @var string
116:      */
117:     public $password;
118: 
119:     /////////////////////////////////////////////////
120:     // PROPERTIES, PRIVATE AND PROTECTED
121:     /////////////////////////////////////////////////
122: 
123:     private $pop_conn;
124:     private $connected;
125:     private $error; //  Error log array
126: 
127:     /**
128:      * Constructor, sets the initial values
129:      * @access public
130:      * @return POP3
131:      */
132: 
133:     public function __construct()
134:     {
135:         $this->pop_conn = 0;
136:         $this->connected = FALSE;
137:         $this->error = NULL;
138:     }
139: 
140:     /**
141:      * Combination of public events - connect, login, disconnect
142:      * @access public
143:      * @param string $host
144:      * @param integer $port
145:      * @param integer $tval
146:      * @param string $username
147:      * @param string $password
148:      */
149: 
150:     public function Authorise($host, $port = FALSE, $tval = FALSE, $username,
151:             $password, $debug_level = 0)
152:     {
153:         $this->host = $host;
154: 
155:         //  If no port value is passed, retrieve it
156:         if ($port == FALSE) {
157:             $this->port = $this->POP3_PORT;
158:         } else {
159:             $this->port = $port;
160:         }
161: 
162:         //  If no port value is passed, retrieve it
163:         if ($tval == FALSE) {
164:             $this->tval = $this->POP3_TIMEOUT;
165:         } else {
166:             $this->tval = $tval;
167:         }
168: 
169:         $this->do_debug = $debug_level;
170:         $this->username = $username;
171:         $this->password = $password;
172: 
173:         //  Refresh the error log
174:         $this->error = NULL;
175: 
176:         //  Connect
177:         $result = $this->Connect($this->host, $this->port, $this->tval);
178: 
179:         if ($result) {
180:             $login_result = $this->Login($this->username, $this->password);
181: 
182:             if ($login_result) {
183:                 $this->Disconnect();
184: 
185:                 return TRUE;
186:             }
187: 
188:         }
189: 
190:         //  We need to disconnect regardless if the login succeeded
191:         $this->Disconnect();
192: 
193:         return FALSE;
194:     }
195: 
196:     /**
197:      * Connect to the POP3 server
198:      * @access public
199:      * @param string $host
200:      * @param integer $port
201:      * @param integer $tval
202:      * @return boolean
203:      */
204: 
205:     public function Connect($host, $port = FALSE, $tval = 30)
206:     {
207:         //  Are we already connected?
208:         if ($this->connected) {
209:             return TRUE;
210:         }
211: 
212:         /*
213:         On Windows this will raise a PHP Warning error if the hostname doesn't exist.
214:         Rather than supress it with @fsockopen, let's capture it cleanly instead
215:          */
216: 
217:         set_error_handler(
218:                 array(
219:                         &$this,
220:                         'catchWarning'
221:                 ));
222: 
223:         //  Connect to the POP3 server
224:         $this->pop_conn = fsockopen($host, //  POP3 Host
225:         $port, //  Port #
226:         $errno, //  Error Number
227:         $errstr, //  Error Message
228:         $tval); //  Timeout (seconds)
229: 
230:         //  Restore the error handler
231:         restore_error_handler();
232: 
233:         //  Does the Error Log now contain anything?
234:         if ($this->error && $this->do_debug >= 1) {
235:             $this->displayErrors();
236:         }
237: 
238:         //  Did we connect?
239:         if ($this->pop_conn == FALSE) {
240:             //  It would appear not...
241:             $this->error = array(
242:                     'error' => "Failed to connect to server $host on port $port",
243:                     'errno' => $errno,
244:                     'errstr' => $errstr
245:             );
246: 
247:             if ($this->do_debug >= 1) {
248:                 $this->displayErrors();
249:             }
250: 
251:             return FALSE;
252:         }
253: 
254:         //  Increase the stream time-out
255: 
256:         //  Check for PHP 4.3.0 or later
257:         if (version_compare(phpversion(), '5.0.0', 'ge')) {
258:             stream_set_timeout($this->pop_conn, $tval, 0);
259:         } else {
260:             //  Does not work on Windows
261:             if (substr(PHP_OS, 0, 3) !== 'WIN') {
262:                 socket_set_timeout($this->pop_conn, $tval, 0);
263:             }
264:         }
265: 
266:         //  Get the POP3 server response
267:         $pop3_response = $this->getResponse();
268: 
269:         //  Check for the +OK
270:         if ($this->checkResponse($pop3_response)) {
271:             //  The connection is established and the POP3 server is talking
272:             $this->connected = TRUE;
273:             return TRUE;
274:         }
275: 
276:     }
277: 
278:     /**
279:      * Login to the POP3 server (does not support APOP yet)
280:      * @access public
281:      * @param string $username
282:      * @param string $password
283:      * @return boolean
284:      */
285: 
286:     public function Login($username = '', $password = '')
287:     {
288:         if ($this->connected == FALSE) {
289:             $this->error = 'Not connected to POP3 server';
290: 
291:             if ($this->do_debug >= 1) {
292:                 $this->displayErrors();
293:             }
294:         }
295: 
296:         if (empty($username)) {
297:             $username = $this->username;
298:         }
299: 
300:         if (empty($password)) {
301:             $password = $this->password;
302:         }
303: 
304:         $pop_username = "USER $username" . $this->CRLF;
305:         $pop_password = "PASS $password" . $this->CRLF;
306: 
307:         //  Send the Username
308:         $this->sendString($pop_username);
309:         $pop3_response = $this->getResponse();
310: 
311:         if ($this->checkResponse($pop3_response)) {
312:             //  Send the Password
313:             $this->sendString($pop_password);
314:             $pop3_response = $this->getResponse();
315: 
316:             if ($this->checkResponse($pop3_response)) {
317:                 return TRUE;
318:             } else {
319:                 return FALSE;
320:             }
321:         } else {
322:             return FALSE;
323:         }
324:     }
325: 
326:     /**
327:      * Disconnect from the POP3 server
328:      * @access public
329:      */
330: 
331:     public function Disconnect()
332:     {
333:         $this->sendString('QUIT');
334: 
335:         fclose($this->pop_conn);
336:     }
337: 
338:     /////////////////////////////////////////////////
339:     //  Private Methods
340:     /////////////////////////////////////////////////
341: 
342:     /**
343:      * Get the socket response back.
344:      * $size is the maximum number of bytes to retrieve
345:      * @access private
346:      * @param integer $size
347:      * @return string
348:      */
349: 
350:     private function getResponse($size = 128)
351:     {
352:         $pop3_response = fgets($this->pop_conn, $size);
353: 
354:         return $pop3_response;
355:     }
356: 
357:     /**
358:      * Send a string down the open socket connection to the POP3 server
359:      * @access private
360:      * @param string $string
361:      * @return integer
362:      */
363: 
364:     private function sendString($string)
365:     {
366:         $bytes_sent = fwrite($this->pop_conn, $string, strlen($string));
367: 
368:         return $bytes_sent;
369:     }
370: 
371:     /**
372:      * Checks the POP3 server response for +OK or -ERR
373:      * @access private
374:      * @param string $string
375:      * @return boolean
376:      */
377: 
378:     private function checkResponse($string)
379:     {
380:         if (substr($string, 0, 3) !== '+OK') {
381:             $this->error = array(
382:                     'error' => "Server reported an error: $string",
383:                     'errno' => 0,
384:                     'errstr' => ''
385:             );
386: 
387:             if ($this->do_debug >= 1) {
388:                 $this->displayErrors();
389:             }
390: 
391:             return FALSE;
392:         } else {
393:             return TRUE;
394:         }
395: 
396:     }
397: 
398:     /**
399:      * If debug is enabled, display the error message array
400:      * @access private
401:      */
402: 
403:     private function displayErrors()
404:     {
405:         echo '<pre>';
406: 
407:         foreach ($this->error as $single_error) {
408:             print_r($single_error);
409:         }
410: 
411:         echo '</pre>';
412:     }
413: 
414:     /**
415:      * Takes over from PHP for the socket warning handler
416:      * @access private
417:      * @param integer $errno
418:      * @param string $errstr
419:      * @param string $errfile
420:      * @param integer $errline
421:      */
422: 
423:     private function catchWarning($errno, $errstr, $errfile, $errline)
424:     {
425:         $this->error[] = array(
426:                 'error' => "Connecting to the POP3 server raised a PHP warning: ",
427:                 'errno' => $errno,
428:                 'errstr' => $errstr
429:         );
430:     }
431: 
432:     //  End of class
433: }
