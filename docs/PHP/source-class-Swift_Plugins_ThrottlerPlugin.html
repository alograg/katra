 89:     /**

 93: 
 94:     public function beforeSendPerformed(Swift_Events_SendEvent $evt)
 95:     {
 96:         $time = $this->getTimestamp();
 97:         if (!isset($this->_start)) {
 98:             $this->_start = $time;
 99:         }
100:         $duration = $time - $this->_start;
101: 
102:         if (self::BYTES_PER_MINUTE == $this->_mode) {
103:             $sleep = $this->_throttleBytesPerMinute($duration);
104:         } else {
105:             $sleep = $this->_throttleMessagesPerMinute($duration);
106:         }
107: 
108:         if ($sleep > 0) {
109:             $this->sleep($sleep);
110:         }
111:     }
