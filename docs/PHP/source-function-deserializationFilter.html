 24: function deserializationFilter(&$amf)
 25: {
 26:     if ($GLOBALS['amfphp']['native'] === TRUE && function_exists('amf_decode')) {
 27:         include_once(AMFPHP_BASE . "amf/io/AMFBaseDeserializer.php");
 28:         include_once(AMFPHP_BASE . "amf/io/AMFBaseSerializer.php");
 29:         $deserializer = new AMFBaseDeserializer($amf->rawData); // deserialize the data
 30:     } else {
 31:         include_once(AMFPHP_BASE . "amf/io/AMFDeserializer.php");
 32:         include_once(AMFPHP_BASE . "amf/io/AMFSerializer.php");
 33:         $deserializer = new AMFDeserializer($amf->rawData); // deserialize the data
 34:     }
 35:     $deserializer->deserialize($amf); // run the deserializer
 36:     //Add some headers
 37:     $headers = $amf->_headerTable;
 38:     if (isset($headers) && is_array($headers)) {
 39:         foreach ($headers as $key => $value) {
 40:             Headers::setHeader($value->name, $value->value);
 41:         }
 42:     }
 43:     //Set as a describe service
 44:     $describeHeader = $amf->getHeader(AMFPHP_SERVICE_BROWSER_HEADER);
 45:     if ($describeHeader !== FALSE) {
 46:         if ($GLOBALS['amfphp']['disableDescribeService']) {
 47:             //Exit
 48:             trigger_error("Service description not allowed", E_USER_ERROR);
 49:             die();
 50:         }
 51:         $bodyCopy = &$amf->getBodyAt(0);
 52:         $bodyCopy->setSpecialHandling('describeService');

 58:  * AuthenticationFilter looks at the credential headers, starts sessions, etc.

 60: function authenticationFilter(&$amf)
 61: {
 62:     $authHeader = $amf->getHeader(AMFPHP_CREDENTIALS_HEADER);
 63:     if ($authHeader !== FALSE
 64:             && $authHeader->value !== AMFPHP_CLEARED_CREDENTIALS) {
 65:         //In PHP5, objects are always pass-by-ref, hence this branch
 66:         if (AMFPHP_PHP5) {
 67:             $bodyCopy = clone ($amf->getBodyAt(0));
 68:         } else {
 69:             $bodyCopy = $amf->getBodyAt(0);
 70:         }
 71:         $uri = $bodyCopy->targetURI;
 72:         $lpos = strrpos($uri, ".");
 73:         $cp = substr($uri, 0, $lpos + 1) . "_authenticate";
 74:         $bodyCopy->targetURI = $cp;
 75:         $bodyCopy->setSpecialHandling('auth');
 76:         $val = $authHeader->value;
 77:         $bodyCopy->setValue($val);
 78:         $amf->addBodyAt(0, $bodyCopy);
 79:         //Make it so that the data will stop being transmitted
 80:         $clearHeader = array(
 81:                 'name' => 'Credentials',
 82:                 'mustUnderstand' => FALSE,
 83:                 'data' => AMFPHP_CLEARED_CREDENTIALS
 84:         );
 85:         $outHeader = new MessageHeader("RequestPersistentHeader", TRUE,
 86:                 $clearHeader);
 87:         $amf->addOutgoingHeader($outHeader);
 88:     }
 89:     $sessionName = session_name();
 90:     if ($sessionName == "" || $sessionName == NULL) {
 91:         //Fix for godaddy not allowing ini_get
 92:         $sessionName = "PHPSESSID";
 93:     }
 94:     session_start();
 95:     $session_id = session_id();
 96:     if (!strpos($_SERVER['QUERY_STRING'], $session_id) !== FALSE) {
 97:         /**
 98:          Instead of trying to guess if using https,
 99:          just use AppendTogGatewayUrl instead
100:          of ReplaceGatewayUrl
101:          */
102:         if (strrpos($_SERVER['QUERY_STRING'], "?") !== FALSE) {
103:             $joint = "&";
104:         } else {
105:             $joint = "?";
106:         }
107:         $outHeader = new MessageHeader("AppendToGatewayUrl", FALSE,
108:                 $joint . $sessionName . "=" . $session_id);
109:         $amf->addOutgoingHeader($outHeader);
110:     }
111: }

113: /**
114:  * Executes each of the bodys
115:  */
116: function batchProcessFilter(&$amf)
117: {
118:     $bodycount = $amf->numBody();
119:     for ($i = 0; $i < $bodycount; $i++) {
120:         $bodyObj = &$amf->getBodyAt($i);
121:         $actions = $GLOBALS['amfphp']['actions'];
122:         foreach ($actions as $key => $action) {
123:             $results = call_user_func($action, $bodyObj);
124:             if ($results === FALSE) {
125:                 break;
126:             }
127:         }
128:     }
129:     $bodycount = $amf->numBody();
130:     for ($i = 0; $i < $bodycount; $i++) {
131:         $bodyObj = &$amf->getBodyAt($i);
132:         if ($bodyObj->getSpecialHandling() == 'auth'
133:                 && $bodyObj->getResults() === NULL) {
134:             $amf->removeBodyAt($i);
135:             break;
136:         }
137:     }
138: }

156:             $amfphp['totalTime'] = microtime_float() - $amfphp['startTime'];
157:             $headerresults[0][] = new ProfilingHeader();
158:         }
159:         //Get the last body in the stack
160:         if (count($headerresults[0]) > 0) {
161:             $body = &$amf->getBodyAt($amf->numBody() - 1);
162:             $headers = new MessageBody(NULL, $body->responseIndex, NULL); // create a new amf body
163:             $headers->responseURI = $body->responseIndex . "/onDebugEvents"; // set the response uri of this body
164:             $headers->setResults($headerresults); // set the results.
165:             $amf->addBodyAt(0, $headers);
166:         }
167:     }
