 51:     /**
 52:      * serialize is the run method of the class.  When serialize is called
 53:      * the AMFObject passed in is read and converted into the amf binary
 54:      * representing the PHP data represented.
 55:      * 
 56:      * @param object $d the AMFObject to serialize
 57:      */
 58:     function serialize(&$amfout)
 59:     {
 60:         $encodeCallback = array(
 61:                 &$this,
 62:                 "encodeCallback"
 63:         );
 64:         $this->writeInt(0); //  write the version ???
 65:         $count = $amfout->numOutgoingHeader();
 66:         $this->writeInt($count); // write header count
 67:         for ($i = 0; $i < $count; $i++) {
 68:             //write headers
 69:             $header = &$amfout->getOutgoingHeaderAt($i);
 70:             $this->writeUTF($header->name);
 71:             $this->writeByte(0);
 72:             $tempBuf = $this->outBuffer;
 73:             $this->outBuffer = "";
 74:             if ($this->native)
 75:                 $this->outBuffer .= amf_encode($header->value,
 76:                         $this->encodeFlags, $encodeCallback);
 77:             else
 78:                 $this->writeData($header->value);
 79:             $tempBuf2 = $this->outBuffer;
 80:             $this->outBuffer = $tempBuf;
 81:             $this->writeLong(strlen($tempBuf2));
 82:             $this->outBuffer .= $tempBuf2;
 83:         }
 84:         $count = $amfout->numBody();
 85:         $this->writeInt($count); // write the body  count
 86:         for ($i = 0; $i < $count; $i++) {
 87:             //write body
 88:             $this->amf0StoredObjects = array();
 89:             $this->storedStrings = array();
 90:             $this->storedObjects = array();
 91:             $this->encounteredStrings = 0;
 92:             $this->storedDefinitions = 0;
 93:             $body = &$amfout->getBodyAt($i);
 94:             $this->currentBody = &$body;

 96:             $this->writeUTF($body->responseTarget); //  write NULL, haven't found another use for this
 97:             $tempBuf = $this->outBuffer;
 98:             $this->outBuffer = "";

326:             return "";

333:         }

336:                 $GLOBALS['amfphp']['outgoingClassMappings'][strtolower(
337:                         $classname)])) {
338:             return $GLOBALS['amfphp']['outgoingClassMappings'][strtolower(
339:                     $classname)];

343:  {
344:             $reflectionClass = new ReflectionClass($classname);
345:             $fileName = $reflectionClass->getFileName();
