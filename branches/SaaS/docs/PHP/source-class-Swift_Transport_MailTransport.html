 11: /**
 12:  * Sends Messages using the mail() function.
 13:  * 
 14:  * It is advised that users do not use this transport if at all possible
 15:  * since a number of plugin features cannot be used in conjunction with this
 16:  * transport due to the internal interface in PHP itself.
 17:  * 
 18:  * The level of error reporting with this transport is incredibly weak, again
 19:  * due to limitations of PHP's internal mail() function.  You'll get an
 20:  * all-or-nothing result from sending.

103:     /**
104:      * Send the given Message.
105:      * 
106:      * Recipient/sender data will be retreived from the Message API.
107:      * The return value is the number of recipients who were accepted for delivery.
108:      * 
109:      * @param Swift_Mime_Message $message
110:      * @param string[] &$failedRecipients to collect failures by-reference
111:      * @return int
112:      */
113: 
114:     public function send(Swift_Mime_Message $message, &$failedRecipients = NULL)
115:     {
116:         $failedRecipients = (array) $failedRecipients;
117: 
118:         if ($evt = $this->_eventDispatcher
119:                 ->createSendEvent($this, $message)) {
120:             $this->_eventDispatcher
121:                     ->dispatchEvent($evt, 'beforeSendPerformed');
122:             if ($evt->bubbleCancelled()) {
123:                 return 0;
124:             }
125:         }
126: 
127:         $count = (count((array) $message->getTo())
128:                 + count((array) $message->getCc())
129:                 + count((array) $message->getBcc()));
130: 
131:         $toHeader = $message->getHeaders()
132:                 ->get('To');
133:         $subjectHeader = $message->getHeaders()
134:                 ->get('Subject');
135: 
136:         $to = $toHeader->getFieldBody();
137:         $subject = $subjectHeader->getFieldBody();
138: 
139:         $reversePath = $this->_getReversePath($message);
140: 
141:         //Remove headers that would otherwise be duplicated
142:         $message->getHeaders()
143:                 ->remove('To');
144:         $message->getHeaders()
145:                 ->remove('Subject');
146: 
147:         $messageStr = $message->toString();
148: 
149:         $message->getHeaders()
150:                 ->set($toHeader);
151:         $message->getHeaders()
152:                 ->set($subjectHeader);
153: 
154:         //Separate headers from body
155:         if (false !== $endHeaders = strpos($messageStr, "\r\n\r\n")) {
156:             $headers = substr($messageStr, 0, $endHeaders) . "\r\n"; //Keep last EOL
157:             $body = substr($messageStr, $endHeaders + 4);
158:         } else {
159:             $headers = $messageStr . "\r\n";
160:             $body = '';
161:         }
162: 
163:         unset($messageStr);
164: 
165:         if ("\r\n" != PHP_EOL) //Non-windows (not using SMTP)
166:  {
167:             $headers = str_replace("\r\n", PHP_EOL, $headers);
168:             $body = str_replace("\r\n", PHP_EOL, $body);
169:         } else //Windows, using SMTP
170:  {
171:             $headers = str_replace("\r\n.", "\r\n..", $headers);
172:             $body = str_replace("\r\n.", "\r\n..", $body);
173:         }
174: 
175:         if ($this->_invoker
176:                 ->mail($to, $subject, $body, $headers,
177:                         sprintf($this->_extraParams, $reversePath))) {
178:             if ($evt) {
179:                 $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
180:                 $evt->setFailedRecipients($failedRecipients);
181:                 $this->_eventDispatcher
182:                         ->dispatchEvent($evt, 'sendPerformed');
183:             }
184:         } else {
185:             $failedRecipients = array_merge($failedRecipients,
186:                     array_keys((array) $message->getTo()),
187:                     array_keys((array) $message->getCc()),
188:                     array_keys((array) $message->getBcc()));
189: 
190:             if ($evt) {
191:                 $evt->setResult(Swift_Events_SendEvent::RESULT_FAILED);
192:                 $evt->setFailedRecipients($failedRecipients);
193:                 $this->_eventDispatcher
194:                         ->dispatchEvent($evt, 'sendPerformed');
195:             }
196: 
197:             $message->generateId();
198: 
199:             $count = 0;
200:         }
201: 
202:         return $count;
203:     }

219:     /** Determine the best-use reverse path for this message */
220: 
221:     private function _getReversePath(Swift_Mime_Message $message)
222:     {
223:         $return = $message->getReturnPath();
224:         $sender = $message->getSender();
225:         $from = $message->getFrom();
226:         $path = NULL;
227:         if (!empty($return)) {
228:             $path = $return;
229:         } elseif (!empty($sender)) {
230:             $keys = array_keys($sender);
231:             $path = array_shift($keys);
232:         } elseif (!empty($from)) {
233:             $keys = array_keys($from);
234:             $path = array_shift($keys);
235:         }
236:         return $path;
237:     }
