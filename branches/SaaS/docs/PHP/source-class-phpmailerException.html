 467:     /**
 468:      * Adds an address to one of the recipient arrays
 469:      * Addresses that have been added already return FALSE, but do not throw exceptions
 470:      * @param string $kind One of 'to', 'cc', 'bcc', 'ReplyTo'
 471:      * @param string $address The email address to send to
 472:      * @param string $name
 473:      * @return boolean TRUE on success, FALSE if address already used or invalid in some way
 474:      * @access private
 475:      */
 476: 
 477:     private function AddAnAddress($kind, $address, $name = '')
 478:     {
 479:         if (!preg_match('/^(to|cc|bcc|ReplyTo)$/', $kind)) {
 480:             echo 'Invalid recipient array: ' . kind;
 481:             return FALSE;
 482:         }
 483:         $address = trim($address);
 484:         $name = trim(preg_replace('/[\r\n]+/', '', $name)); //Strip breaks and trim
 485:         if (!self::ValidateAddress($address)) {
 486:             $this->SetError($this->Lang('invalid_address') . ': ' . $address);
 487:             if ($this->exceptions) {
 488:                 throw new phpmailerException(
 489:                         $this->Lang('invalid_address') . ': ' . $address);
 490:             }
 491:             echo $this->Lang('invalid_address') . ': ' . $address;
 492:             return FALSE;
 493:         }
 494:         if ($kind != 'ReplyTo') {
 495:             if (!isset($this->all_recipients[strtolower($address)])) {
 496:                 array_push($this->$kind,
 497:                         array(
 498:                                 $address,
 499:                                 $name
 500:                         ));
 501:                 $this->all_recipients[strtolower($address)] = TRUE;
 502:                 return TRUE;
 503:             }
 504:         } else {
 505:             if (!array_key_exists(strtolower($address), $this->ReplyTo)) {
 506:                 $this->ReplyTo[strtolower($address)] = array(
 507:                         $address,
 508:                         $name
 509:                 );
 510:                 return TRUE;
 511:             }
 512:         }
 513:         return FALSE;
 514:     }

 549:     /**

 551:      * Static so it can be used without instantiation
 552:      * Tries to use PHP built-in validator in the filter extension (from PHP 5.2), falls back to a reasonably competent regex validator
 553:      * Conforms approximately to RFC2822
 554:      * @link http://www.hexillion.com/samples/#Regex Original pattern found here
 555:      * @param string $address The email address to check
 556:      * @return boolean
 557:      * @static
 558:      * @access public

 563:         if (function_exists('filter_var')) { //Introduced in PHP 5.2
 564:             if (filter_var($address, FILTER_VALIDATE_EMAIL) === FALSE) {
 565:                 return FALSE;
 566:             } else {

 568:             }
 569:         } else {

 580:     /**
 581:      * Creates message and assigns Mailer. If the message is
 582:      * not sent successfully then it returns FALSE.  Use the ErrorInfo
 583:      * variable to view description of the error.
 584:      * @return bool
 585:      */
 586: 
 587:     public function Send()
 588:     {
 589:         try {
 590:             if ((count($this->to) + count($this->cc) + count($this->bcc)) < 1) {
 591:                 throw new phpmailerException($this->Lang('provide_address'),
 592:                         self::STOP_CRITICAL);
 593:             }
 594: 
 595:             // Set whether the message is multipart/alternative
 596:             if (!empty($this->AltBody)) {
 597:                 $this->ContentType = 'multipart/alternative';
 598:             }
 599: 
 600:             $this->error_count = 0; // reset errors
 601:             $this->SetMessageType();
 602:             $header = $this->CreateHeader();
 603:             $body = $this->CreateBody();
 604: 
 605:             if (empty($this->Body)) {
 606:                 throw new phpmailerException($this->Lang('empty_message'),
 607:                         self::STOP_CRITICAL);
 608:             }
 609: 
 610:             // digitally sign with DKIM if enabled
 611:             if ($this->DKIM_domain && $this->DKIM_private) {
 612:                 $header_dkim = $this->DKIM_Add($header, $this->Subject, $body);
 613:                 $header = str_replace("\r\n", "\n", $header_dkim) . $header;
 614:             }
 615: 
 616:             // Choose the mailer and send through it
 617:             switch ($this->Mailer) {
 618:                 case 'sendmail':
 619:                     return $this->SendmailSend($header, $body);
 620:                 case 'smtp':
 621:                     return $this->SmtpSend($header, $body);
 622:                 default:
 623:                     return $this->MailSend($header, $body);
 624:             }
 625: 
 626:         } catch (phpmailerException $e) {
 627:             $this->SetError($e->getMessage());
 628:             if ($this->exceptions) {
 629:                 throw $e;
 630:             }
 631:             echo $e->getMessage() . "\n";
 632:             return FALSE;
 633:         }
 634:     }

1127:     /**
1128:      * Finds last character boundary prior to maxLength in a utf-8
1129:      * quoted (printable) encoded string.
1130:      * Original written by Colin Brown.
1131:      * @access public
1132:      * @param string $encodedText utf-8 QP text
1133:      * @param int    $maxLength   find last character boundary prior to this length
1134:      * @return int
1135:      */
1136: 
1137:     public function UTF8CharBoundary($encodedText, $maxLength)
1138:     {
1139:         $foundSplitPos = FALSE;
1140:         $lookBack = 3;
1141:         while (!$foundSplitPos) {
1142:             $lastChunk = substr($encodedText, $maxLength - $lookBack, $lookBack);
1143:             $encodedCharPos = strpos($lastChunk, "=");
1144:             if ($encodedCharPos !== FALSE) {
1145:                 // Found start of encoded character byte within $lookBack block.
1146:                 // Check the encoded byte value (the 2 chars after the '=')
1147:                 $hex = substr($encodedText,
1148:                         $maxLength - $lookBack + $encodedCharPos + 1, 2);
1149:                 $dec = hexdec($hex);
1150:                 if ($dec < 128) { // Single byte character.
1151:                 // If the encoded char was found at pos 0, it will fit
1152:                 // otherwise reduce maxLength to start of the encoded char
1153:                     $maxLength = ($encodedCharPos == 0) ? $maxLength
1154:                             : $maxLength - ($lookBack - $encodedCharPos);
1155:                     $foundSplitPos = TRUE;
1156:                 } elseif ($dec >= 192) { // First byte of a multi byte character
1157:                 // Reduce maxLength to split at start of character
1158:                     $maxLength = $maxLength - ($lookBack - $encodedCharPos);
1159:                     $foundSplitPos = TRUE;
1160:                 } elseif ($dec < 192) { // Middle byte of a multi byte character, look further back
1161:                     $lookBack += 3;
1162:                 }
1163:             } else {
1164:                 // No encoded character found
1165:                 $foundSplitPos = TRUE;
1166:             }
1167:         }
1168:         return $maxLength;
1169:     }

1730: 
1731:         switch (strtolower($position)) {
1732:             case 'phrase':

1808:      * without breaking lines within a character.
1809:      * Adapted from a function by paravoid at http://uk.php.net/manual/en/function.mb-encode-mimeheader.php
1810:      * @access public
1811:      * @param string $str multi-byte text to wrap encode
1812:      * @return string
1813:      */
1814: 
1815:     public function Base64EncodeWrapMB($str)
1816:     {
1817:         $start = "=?" . $this->CharSet . "?B?";
1818:         $end = "?=";
1819:         $encoded = "";
1820: 
1821:         $mb_length = mb_strlen($str, $this->CharSet);
1822:         // Each line must have length <= 75, including $start and $end
1823:         $length = 75 - strlen($start) - strlen($end);
1824:         // Average multi-byte ratio
1825:         $ratio = $mb_length / strlen($str);
1826:         // Base64 has a 4:3 ratio
1827:         $offset = $avgLength = floor($length * $ratio * .75);
1828: 
1829:         for ($i = 0; $i < $mb_length; $i += $offset) {
1830:             $lookBack = 0;
1831: 
1832:             do {
1833:                 $offset = $avgLength - $lookBack;
1834:                 $chunk = mb_substr($str, $i, $offset, $this->CharSet);
1835:                 $chunk = base64_encode($chunk);
1836:                 $lookBack++;
1837:             } while (strlen($chunk) > $length);
1838: 
1839:             $encoded .= $chunk . $this->LE;
1840:         }
1841: 
1842:         // Chomp the last linefeed
1843:         $encoded = substr($encoded, 0, -strlen($this->LE));
1844:         return $encoded;
