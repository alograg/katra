129:     /**
130:      * writeArray first deterines if the PHP array contains all numeric indexes
131:      * or a mix of keys.  Then it either writes the array code (0x0A) or the
132:      * object code (0x03) and then the associated data.
133:      * 
134:      * @param array $d The php array
135:      */
136:     function writeArray($d)
137:     {
138:         if ($this->writeReferenceIfExists($d)) {
139:             return;
140:         }
141: 
142:         $numeric = array(); // holder to store the numeric keys
143:         $string = array(); // holder to store the string keys
144:         $len = count($d); // get the total number of entries for the array
145:         $largestKey = -1;
146:         foreach ($d as $key => $data) { // loop over each element
147:             if (is_int($key) && ($key >= 0)) { // make sure the keys are numeric
148:                 $numeric[$key] = $data; // The key is an index in an array
149:                 $largestKey = max($largestKey, $key);
150:             } else {
151:                 $string[$key] = $data; // The key is a property of an object
152:             }
153:         }
154:         $num_count = count($numeric); // get the number of numeric keys
155:         $str_count = count($string); // get the number of string keys
156: 
157:         if (($num_count > 0 && $str_count > 0)
158:                 || ($num_count > 0 && $largestKey != $num_count - 1)) { // this is a mixed array
159: 
160:             $this->writeByte(8); // write the mixed array code
161:             $this->writeLong($num_count); // write  the count of items in the array
162:             $this->writeObjectFromArray($numeric + $string); // write the numeric and string keys in the mixed array
163:         } else if ($num_count > 0) { // this is just an array
164: 
165:             $num_count = count($numeric); // get the new count
166: 
167:             $this->writeByte(10); // write  the array code
168:             $this->writeLong($num_count); // write  the count of items in the array
169:             for ($i = 0; $i < $num_count; $i++) { // write all of the array elements
170:                 $this->writeData($numeric[$i]);
171:             }
172:         } else if ($str_count > 0) { // this is an object
173:             $this->writeByte(3); // this is an  object so write the object code
174:             $this->writeObjectFromArray($string); // write the object name/value pairs
175:         } else { //Patch submitted by Jason Justman
176: 
177:             $this->writeByte(10); // make this  an array still
178:             $this->writeInt(0); //  give it 0 elements
179:             $this->writeInt(0); //  give it an element pad, this looks like a bug in Flash,
180:             //but keeps the next alignment proper
181:         }

296:     /**
297:      * writeRecordSet is the abstracted method to write a custom class recordset object.
298:      * Any recordset from any datasource can be written here, it just needs to be properly formatted
299:      * beforehand.
300:      *
301:      * This was unrolled with at the expense of readability for a 
302:      * 10 fold increase in speed in large recordsets
303:      * 
304:      * @param object $rs The formatted RecordSet object
305:      */
306: 
307:     function writeRecordSet(&$rs)
308:     {
309:         //Low-level everything here to make things faster
310:         //This is the bottleneck of AMFPHP, hence the attention in making things faster
311:         if ($this->writeReferenceIfExists($rs)) {
312:             return;
313:         }
314: 
315:         $ob = "";
316:         $data = $rs->rows;
317: 
318:         if ($GLOBALS['amfphp']['encoding'] == 'amf0') {
319: 
320:             $this->writeByte(16); // write  the custom class code
321:             $this->writeUTF("RecordSet"); // write  the class name
322:             $this->writeUTF("serverInfo");
323: 
324:             //Start writing inner object
325:             $this->writeByte(3); // this is an  object so write the object code
326: 
327:             //Write total count
328:             $this->writeUTF("totalCount");
329:             $this->writeNumber($rs->getRowCount());
330: 
331:             //Write initial data
332:             $this->writeUTF("initialData");
333: 
334:             //Inner numeric array
335:             $colnames = $rs->columns;
336: 
337:             $num_count = count($rs->rows);
338:             $this->writeByte(10); // write  the mixed array code
339:             $this->writeLong($num_count); // write  the count of items in the array
340: 
341:             //Allow recordsets to create their own serialized data, which is faster
342:             //since the recordset array is traversed only once
343:             $numcols = count($colnames);
344: 
345:             $ob = "";
346:             $be = $this->isBigEndian;
347:             $fc = pack('N', $numcols);
348: 
349:             for ($i = 0; $i < $num_count; $i++) {
350:                 // write all of the array elements
351:                 $ob .= "\12" . $fc;
352: 
353:                 for ($j = 0; $j < $numcols; $j++) { // write all of the array elements
354: 
355:                     $d = $data[$i][$j];
356:                     if (is_string($d)) { // type as string
357:                         $os = $this->rsCharsetHandler
358:                                 ->transliterate($d);
359:                         //string flag, string length, and string
360:                         $ob .= "\2" . pack('n', strlen($os)) . $os;
361:                     } elseif (is_float($d) || is_int($d)) { // type as double
362:                         $ob .= "\0";
363:                         $b = pack('d', $d); // pack the bytes
364:                         if ($be) { // if we are a big-endian processor
365:                             $r = strrev($b);
366:                         } else { // add the bytes to the output
367:                             $r = $b;
368:                         }
369:                         $ob .= $r;
370:                     } elseif (is_bool($d)) { //type as bool
371:                         $ob .= "\1";
372:                         $ob .= pack('c', $d);
373:                     } elseif (is_null($d)) { // NULL
374:                         $ob .= "\5";
375:                     }
376:                 }
377:             }
378:             $this->outBuffer .= $ob;
379: 
380:             //Write cursor
381:             $this->writeUTF("cursor");
382:             $this->writeNumber(1);
383: 
384:             //Write service name
385:             $this->writeUTF("serviceName");
386:             $this->writeString("PageAbleResult");
387: 
388:             //Write column names
389:             $this->writeUTF("columnNames");
390:             $this->writePlainArray($colnames, 'string');
391: 
392:             //Write version number
393:             $this->writeUTF("version");
394:             $this->writeNumber(1);
395: 
396:             //Write id
397:             $this->writeUTF("id");
398:             $this->writeString($rs->getID());
399: 
400:             //End inner serverInfo object
401:             $this->writeInt(0); //  write the end object flag 0x00, 0x00, 0x09
402:             $this->writeByte(9);
403: 
404:             //End outer recordset object
405:             $this->writeInt(0); //  write the end object flag 0x00, 0x00, 0x09
406:             $this->writeByte(9);
407: 
408:             $this->paging = -1;
409:         } else {
410:             $numObjects = 0;
411:             $this->writeAmf3ArrayCollectionPreamble();
412: 
413:             //Amf3 array code
414:             $this->writeByte(0x09);
415:             $numObjects++;
416: 
417:             $numRows = count($rs->rows);
418:             $toPack = 2 * $numRows + 1;
419: 
420:             //Write the number of rows
421:             $this->writeAmf3Int($toPack);
422: 
423:             //No string keys in this array
424:             $this->writeByte(0x01);
425: 
426:             $numCols = count($rs->columns);
427: 
428:             $columnStringOffsets = array();
429:             if ($numRows > 0) {
430:                 $j = 0;
431:                 $colNames = array();
432:                 $rows = $rs->rows;
433: 
434:                 foreach ($rows as $key => $line) {
435: 
436:                     //Usually we don't use class defs in the serializer since we don't 
437:                     //have sealed objects in php, but for recordsets we do use them
438:                     //since they are well suited for what we have to do (the same keys
439:                     //across all objects)
440:                     if ($key == 0) {
441:                         $this->outBuffer .= "\12";
442:                         $this->writeAmf3Int($numCols << 4 | 3);
443:                         $this->outBuffer .= "\1";
444:                         foreach ($rs->columns as $key => $val) {
445:                             $this->writeAmf3String($val);
446:                         }
447:                         $defOffset = $this->getAmf3Int(
448:                                         ($this->storedDefinitions) << 2 | 1);
449:                         $this->storedDefinitions++;
450:                     } else {
451:                         $this->outBuffer .= "\12" . $defOffset;
452:                     }
453:                     $numObjects++;
454: 
455:                     for ($i = 0; $i < $numCols; $i++) {
456:                         //Write the col name
457:                         $value = $line[$i];
458:                         if (is_string($value)) {
459:                             $this->outBuffer .= "\6";
460:                             $value = $this->rsCharsetHandler
461:                                     ->transliterate($value);
462:                             $this->writeAmf3String($value, TRUE);
463:                         } elseif (is_int($value)) { //int
464:                             $this->writeAmf3Number($value);
465:                         } elseif (is_float($value)) { //double
466:                             $this->outBuffer .= "\5";
467:                             $b = pack("d", $value); // pack the bytes
468:                             if ($this->isBigEndian) { // if we are a big-endian processor
469:                                 $r = strrev($b);
470:                             } else { // add the bytes to the output
471:                                 $r = $b;
472:                             }
473: 
474:                             $this->outBuffer .= $r;
475:                         } elseif (is_bool($value)) {
476:                             $this->outBuffer .= $value ? "\3" : "\2";
477:                         } else {
478:                             $this->outBuffer .= "\1"; //null
479:                         }
480:                     }
481:                     //End object
482:                 }
483:             }
484: 
485:             //Add fake objects to make sure the object counter still works
486:             for ($i = 0; $i < $numObjects; $i++) {
487:                 $this->storedObjects[] = "";
488:             }
489:         }

492:     /**
493:      * writeData checks to see if the type was declared and then either
494:      * auto negotiates the type or relies on the user defined type to
495:      * serialize the data into amf
496:      *
497:      * Note that autoNegotiateType was eliminated in order to tame the 
498:      * call stack which was getting huge and was causing leaks
499:      *
500:      * manualType allows the developer to explicitly set the type of
501:      * the returned data.  The returned data is validated for most of the
502:      * cases when possible.  Some datatypes like xml and date have to
503:      * be returned this way in order for the Flash client to correctly serialize them
504:      * 
505:      * recordsets appears top on the list because that will probably be the most
506:      * common hit in this method.  Followed by the
507:      * datatypes that have to be manually set.  Then the auto negotiatable types last.
508:      * The order may be changed for optimization.
509:      * 
510:      * @param mixed $d The data
511:      * @param string $type The optional type
512:      */
513:     function writeData(&$d)
514:     {
515:         if (is_int($d) || is_float($d)) { // double
516:             $this->writeNumber($d);
517:             return;
518:         } elseif (is_string($d)) { // string
519:             $this->writeString($d);
520:             return;
521:         } elseif (is_bool($d)) { // boolean
522:             $this->writeBoolean($d);
523:             return;
524:         } elseif (is_null($d)) { // NULL
525:             $this->writeNull();
526:             return;
527:         } elseif ($GLOBALS['amfphp']['encoding'] == 'amf3') {
528:             $this->writeByte(0x11);
529:             $this->writeAmf3Data($d);
530:             return;
531:         } elseif (is_array($d)) { // array
532:             $this->writeArray($d);
533:             return;
534:         } elseif (is_resource($d)) { // resource
535:             $type = get_resource_type($d);
536:             list($type, $subtype) = $this->sanitizeType($type);
537:         } elseif (is_object($d)) {
538:             $className = strtolower(get_class($d));
539:             if (array_key_exists($className, $this->resourceObjects)) {
540:                 $type = "__RECORDSET__";
541:                 $subtype = $this->resourceObjects[strtolower(get_class($d))];
542:             } else if (AMFPHP_PHP5 && $className == 'domdocument') {
543:                 $this->writeXML($d->saveXml());
544:                 return;
545:             } else if (!AMFPHP_PHP5 && $className == 'domdocument') {
546:                 $this->writeXML($d->dump_mem());
547:                 return;
548:             } elseif ($className == "simplexmlelement") {
549:                 $this->writeXML($d->asXML());
550:                 return;
551:             } else if ($className == 'stdclass' && !isset($d->_explicitType)) {
552:                 $this->writeAnonymousObject($d);
553:                 return;
554:             }
555:             //Fix for PHP5 overriden ArrayAccess and ArrayObjects with an explcit type
556:  elseif ((is_a($d, 'ArrayAccess') || is_a($d, 'ArrayObject'))
557:                     && !isset($d->_explicitType)) {
558:                 $this->writeArray($d);
559:                 return;
560:             } else {
561:                 $this->writeTypedObject($d);
562:                 return;
563:             }
564:         } else {
565:             $type = gettype($d);
566:         }
567: 
568:         switch ($type) {
569:             case "mysql link":
570:                 $this->writeString($d);
571:                 break;
572:             case "__RECORDSET__":
573:                 $classname = $subtype . "Adapter"; // full class name
574:                 $includeFile = include_once(AMFPHP_BASE . "shared/adapters/"
575:                         . $classname . ".php"); // try to load the recordset library from the sql folder
576:                 if (!$includeFile) {
577:                     trigger_error(
578:                             "The recordset filter class " . $classname
579:                                     . " was not found", E_USER_ERROR);
580:                 }
581:                 $recordSet = new $classname($d); // returns formatted recordset
582:                 $this->writeRecordSet($recordSet); // writes the recordset formatted for Flash
583:                 break;
584:             default:
585:             // non of the above so lets assume its a Custom Class thats defined in the client
586:                 $this->writeTypedObject($unsanitizedType, $d);
587:                 // trigger_error("Unsupported Datatype");
588:                 break;
589:         }
590:     }
